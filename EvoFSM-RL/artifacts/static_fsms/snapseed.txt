STRATEGIES:
  OPEN_APP:
    preconditions: Device on home screen
    steps:
      1. Invoke open_app with name 'Snapseed'
      2. Verify snapseed_main visible
      3. If not visible, open app drawer via scroll and tap the Snapseed icon
    success_signal: snapseed_main screen shows 'Tap anywhere to open a photo' and OPEN button
    fallback: navigate_home then open_app again; if still fails, scroll app drawer and tap icon directly
  ENABLE_DARK_THEME:
    preconditions: snapseed_main
    steps:
      1. Click More options (3-dot)
      2. Click Settings
      3. Click Dark theme toggle
    success_signal: Dark theme toggle blue/on and UI chrome becomes dark
    fallback: If dropdown doesn't open, retry click on More options
  SET_IMAGE_SIZING:
    preconditions: snapseed_main
    steps:
      1. Click More options
      2. Click Settings
      3. Click Image sizing row
      4. Pick desired size (e.g., '2000 px') in dialog
    success_signal: Settings page shows chosen size under Image sizing
    fallback: Reopen Settings and click Image sizing again
  SET_FORMAT_QUALITY:
    preconditions: snapseed_main
    steps:
      1. Click More options
      2. Click Settings
      3. Click Format and quality row
      4. Pick desired option (e.g., 'JPG 100%')
    success_signal: Settings page shows chosen format under Format and quality
    fallback: Retry clicking the setting row if dialog doesn't appear
  OPEN_PHOTO:
    preconditions: snapseed_main
    steps:
      1. Click OPEN or tap anywhere prompt
      2. In system picker choose a photo
      3. Wait for editor to load
    success_signal: editor_view (Looks/Tools/Export tabs) visible
    fallback: If Could not load photo appears, dismiss (OK) and choose a different file; if no media, use Gallery > Add folder to include a folder with images
DEAD_ENDS:
  at could_not_load_photo_error: selecting thumbnails repeatedly fails  [Every available Recent image returned 'Could not load photo'; retrying the same or other thumbnails does not succeed. Photo-dependent tasks become infeasible in this environment.]
  at gallery_no_media: Add folder then OK fails  [Folders contain only .mp4 files, not images; folder selection does not produce loadable images for Snapseed.]
  at photo_picker: navigate_back/navigate_home loops fails  [Repeatedly returning to home and reopening Snapseed does not bypass the photo loading error.]

# LAYER 2: GENERIC  (transferable, app-agnostic)

CATEGORY: LAUNCH_APP
  precondition: Device idle on a top-level launcher surface
  abstract_steps:
    1. Invoke a direct app-launch mechanism referencing the target application by its user-visible label
    2. Verify the expected app landing screen is shown
    3. If not shown, access the full app list and select the target application icon
  failure_modes:
    - 'Direct launch no-ops silently when target cannot be resolved'
    - 'A different application with a similar name is launched instead'
    - 'Repeated retries without changing strategy'
  verification_checklist:
    - 'Landing screen of target application visible'
    - 'Characteristic UI elements of target application detected'

CATEGORY: TOGGLE_SETTING
  precondition: Inside the target application
  abstract_steps:
    1. Reveal the overflow/options menu from the top-level UI
    2. Navigate to the configuration area
    3. Locate the desired toggle
    4. Tap the toggle to change state
  failure_modes:
    - 'Overflow menu missed'
    - 'Toggle already in desired state and action has no visible effect'
    - 'Control hidden behind a subsection'
  verification_checklist:
    - 'Toggle visually reflects desired state (on/off indicator)'
    - 'Associated UI updates (e.g., theme changes)'

CATEGORY: SELECT_FROM_OPTION_LIST
  precondition: A configuration row that presents a single-choice dialog is reachable
  abstract_steps:
    1. Enter the configuration area
    2. Tap the configuration row that presents a choice dialog
    3. Tap the target option
    4. Dismiss dialog if needed
  failure_modes:
    - 'Wrong option clicked'
    - 'Dialog closes without selection if tap misses'
    - 'Row not scrolled into view'
  verification_checklist:
    - 'Configuration row shows the new selected value'
    - 'Dialog closes after selection'

CATEGORY: SELECT_CONTENT_ITEM
  precondition: Application requires selecting a piece of external content (e.g., a file/photo) to proceed
  abstract_steps:
    1. Trigger the content picker from the application
    2. Browse available items or switch source
    3. Choose a candidate item
    4. Wait for the application to load it
  failure_modes:
    - 'Item cannot be loaded (corrupted/unsupported)'
    - 'No items available in current source'
    - 'Permission not granted to access storage'
  verification_checklist:
    - 'Editor/consumer view of the selected item is presented'
    - 'No error dialog present'

CATEGORY: RECOVER_FROM_BLOCKING_DIALOG
  precondition: A modal dialog (error or info) is blocking interaction
  abstract_steps:
    1. Dismiss the modal via its affirmative/close button
    2. Reassess state
    3. Take an alternative path instead of repeating the exact action
  failure_modes:
    - 'Looping: dismissing and immediately retrying the same failing action'
    - 'Mistaking modal buttons for other controls'
  verification_checklist:
    - 'Modal no longer present'
    - 'Underlying screen interactive again'

CATEGORY: ABANDON_TASK_WHEN_BLOCKED
  precondition: Repeated alternative attempts fail due to environment-level issue
  abstract_steps:
    1. Detect a repeating failure signature
    2. Try at least one alternative source/option
    3. If still failing, mark task infeasible rather than loop
  failure_modes:
    - 'Endless retry loops consuming steps'
    - 'False positive marking feasible tasks infeasible too early'
  verification_checklist:
    - 'At least one distinct alternative tried'
    - 'Clear evidence the blocker is environmental, not a UI navigation mistake'
