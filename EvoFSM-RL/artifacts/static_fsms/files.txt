# LAYER 2: GENERIC  (transferable, app-agnostic)
# ═══════════════════════════════════════════════
CATEGORY: NAVIGATE_TO_ITEM
  precondition: App launched; target item location known or inferable from path
  abstract_steps:
    1. Open the content browsing app directly rather than via indirect system entry points
    2. Open the roots/drawer to see top-level locations
    3. Enter the correct top-level volume
    4. Drill into subfolders matching the target path
    5. Scroll within the folder if the item is not immediately visible
    6. Optionally use Search to locate the item by name
  failure_modes:
    - 'Treating category filters (Images/Audio/etc.) as folder navigation'
    - 'Staying in a default landing folder instead of switching volumes'
    - 'Looping through unrelated system screens that do not expose raw folders'
  verification_checklist:
    - 'Header shows the intended folder name'
    - 'Target item appears in the listing'

CATEGORY: DELETE_ENTRY
  precondition: Target item visible in its containing view
  abstract_steps:
    1. Long-press the item to enter selection mode
    2. Trigger the deletion action from the action bar
    3. Confirm the deletion in the confirmation dialog
  failure_modes:
    - 'Selecting the wrong item with a similar name'
    - 'Dismissing the confirmation dialog instead of confirming'
    - 'Attempting to perform the deletion from a summary/cleanup screen instead of the primary content browsing view'
  verification_checklist:
    - 'Transient deletion feedback appears'
    - 'Item no longer present in the containing view'

CATEGORY: RELOCATE_ENTRY
  precondition: Target item visible; destination folder exists and is known
  abstract_steps:
    1. Long-press the item to enter selection mode
    2. Open the overflow/more-options menu
    3. Choose the relocation action
    4. In the destination picker, navigate to the exact destination folder (switching roots/volumes as needed)
    5. Confirm the relocation from within the destination folder
  failure_modes:
    - 'Confirming the relocation while the picker is still at the default location'
    - 'Choosing a duplication action or a metadata-viewing action by mistake'
    - 'Not switching to the correct top-level volume in the picker'
  verification_checklist:
    - 'Transient relocation feedback appears'
    - 'Item gone from source folder'
    - 'Item present in destination folder'

CATEGORY: QUERY_METADATA
  precondition: Item or folder visible
  abstract_steps:
    1. Select the item or open overflow menu
    2. Choose the details/properties option
    3. Read the displayed metadata
  failure_modes:
    - 'Confusing the details screen with an actionable screen'
  verification_checklist:
    - 'Details page shows expected attributes (type, modified date, size/count)'
